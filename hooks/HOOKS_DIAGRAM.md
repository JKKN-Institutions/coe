## Hook Composition Examples

### Example 1: Smart Table Component
```typescript
function SmartTable() {
  // Data
  const { items, loading } = useCRUD({ apiEndpoint: '/api/courses' })

  // Processing
  const { filteredData } = useAdvancedFilter(items)
  const { sortedData } = useAdvancedSort(filteredData)

  // Presentation
  const { paginatedData, ...tableState } = useDataTable({ data: sortedData })

  // UI
  const isMobile = useIsMobile()
  const { isOpen, openForEdit } = useSheet()

  return <Table data={paginatedData} {...tableState} />
}
```

### Example 2: Examination Dashboard
```typescript
function ExamDashboard() {
```

### Example 3: Form with Validation
```typescript
function CourseForm() {
  // Form state
  const [formData, setFormData] = useState(initialData)

  // Validation
  const { errors, validate } = useFormValidation(validationRules)

  // API
  const { createItem, loading } = useCRUD({ apiEndpoint: '/api/courses' })

  // UI
  const { isOpen, close } = useSheet()

  const handleSubmit = async () => {
    if (!validate(formData)) return
    await createItem(formData)
    close()
  }

  return <Form data={formData} errors={errors} onSubmit={handleSubmit} />
}
```
